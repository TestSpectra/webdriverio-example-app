Storage

Examples of managing localStorage and sessionStorage in WebDriverIO, for a full reference of commands, go to webdriver.io

Clear Local Storage

To manage localStorage, use browser.execute() to access it directly.

// Click button to populate storage
const btn = await $('.ls-btn')
await btn.click()

// Check localStorage values
const storage = await browser.execute(() => localStorage)
await expect(storage.prop1).toBe('red')
await expect(storage.prop2).toBe('blue')
await expect(storage.prop3).toBe('magenta')

// Clear all localStorage
await browser.execute(() => localStorage.clear())
const cleared = await browser.execute(() => localStorage)
await expect(Object.keys(cleared)).toHaveLength(0)

// Populate again
await btn.click()

// Clear specific key
await browser.execute(() => localStorage.removeItem('prop1'))
const partial = await browser.execute(() => localStorage)
await expect(partial.prop1).toBeUndefined()
await expect(partial.prop2).toBe('blue')
await expect(partial.prop3).toBe('magenta')

Get Local Storage

To get all data in localStorage, use browser.execute() to access it.

// Click button to populate storage
const btn = await $('.ls-btn')
await btn.click()

// Get all localStorage data
const storage = await browser.execute(() => {
  return Object.fromEntries(Object.entries(localStorage))
})

await expect(storage).toEqual({
  prop1: 'red',
  prop2: 'blue',
  prop3: 'magenta',
})

Clear All Local Storage

To clear all localStorage data, use browser.execute().

// Click button to populate storage
const btn = await $('.ls-btn')
await btn.click()

// Clear all localStorage
await browser.execute(() => localStorage.clear())

// Verify it's cleared
const storage = await browser.execute(() => localStorage.length)
await expect(storage).toBe(0)

Get Session Storage

To get all data in sessionStorage, use browser.execute() to access it.

// Click button to populate storage
const btn = await $('.ls-btn')
await btn.click()

// Get all sessionStorage data
const storage = await browser.execute(() => {
  return Object.fromEntries(Object.entries(sessionStorage))
})

await expect(storage).toEqual({
  prop4: 'cyan',
  prop5: 'yellow',
  prop6: 'black',
})

Clear All Session Storage

To clear all sessionStorage data, use browser.execute().

// Click button to populate storage
const btn = await $('.ls-btn')
await btn.click()

// Clear all sessionStorage
await browser.execute(() => sessionStorage.clear())

// Verify it's cleared
const storage = await browser.execute(() => sessionStorage.length)
await expect(storage).toBe(0)
