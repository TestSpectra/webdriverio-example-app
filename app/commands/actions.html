Actions

Examples of actions being performed on DOM elements in Webdriver.io, for a full reference of commands, go to webdriver.io

.setValue()

To type into a DOM element, use the .setValue() command.

const input = await $('.action-email')
await input.setValue('fake@email.com')
await expect(input).toHaveValue('fake@email.com')

// Send special characters
const input = await $('.action-email')
await input.addValue(['ArrowLeft', 'ArrowRight', 'ArrowUp', 'ArrowDown'])
await input.addValue(['Delete', 'Control', 'a', 'Backspace'])

// .setValue() with key modifiers
await input.addValue(['Alt', 'Alt'])
await input.addValue(['Control', 'Control'])
await input.addValue(['Meta', 'Meta'])
await input.addValue('Shift')

// For a slow typing effect, you can use a loop with a delay
const slowEmail = 'slow.typing@email.com'
for (const char of slowEmail) {
  await input.addValue(char)
  await browser.pause(100)
}
await expect(input).toHaveValue(slowEmail)

const disabledInput = await $('.action-disabled')
// Webdriver.io does not have a force option, you would need to enable the element first
await browser.execute((el) => el.disabled = false, disabledInput)
await disabledInput.setValue('disabled error checking')
await expect(disabledInput).toHaveValue('disabled error checking')

const input = await $('.action-focus')

await expect(input).toHaveClass('focus')

.clearValue()

To blur on a DOM element, use the .clearValue() command.

const input = await $('.action-blur')
await input.setValue('About to blur')
await input.clearValue()
await expect(input).toHaveClass('error')
await expect(input.prev()).toHaveAttr('style', 'color: red;')

.clearValue()

To clear on a DOM element, use the .clearValue() command.

const input = await $('.action-clear')
await input.setValue('Clear this text')
await expect(input).toHaveValue('Clear this text')
await input.clearValue()
await expect(input).toHaveValue('')

.click()

To submit a form, you can use the .click() command on the submit button.

const form = await $('.action-form')
const input = await form.$('[type="text"]')
await input.setValue('HALFOFF')

const submitButton = await form.$('button[type="submit"]')
await submitButton.click()

const nextElement = await form.next()
await expect(nextElement).toHaveTextContaining('Your form has been submitted!')

.click()

To click a DOM element, use the .click() command.

const button = await $('.action-btn')
await button.click()

// You can click on 9 specific positions of an element:
//  -----------------------------------
// | topLeft        top       topRight |
// |                                   |
// |                                   |
// |                                   |
// | left          center        right |
// |                                   |
// |                                   |
// |                                   |
// | bottomLeft   bottom   bottomRight |
//  -----------------------------------

// clicking in the center of the element is the default
cy.get('#action-canvas').click()

cy.get('#action-canvas').click('topLeft')
cy.get('#action-canvas').click('top')
cy.get('#action-canvas').click('topRight')
cy.get('#action-canvas').click('left')
cy.get('#action-canvas').click('right')
cy.get('#action-canvas').click('bottomLeft')
cy.get('#action-canvas').click('bottom')
cy.get('#action-canvas').click('bottomRight')

// .click() accepts an x and y coordinate
// that controls where the click occurs :)

cy.get('#action-canvas')
cy.get('#action-canvas').click(80, 75) // click 80px on x coord and 75px on y coord
cy.get('#action-canvas').click(170, 75)
cy.get('#action-canvas').click(80, 165)
cy.get('#action-canvas').click(100, 185)
cy.get('#action-canvas').click(125, 190)
cy.get('#action-canvas').click(150, 185)
cy.get('#action-canvas').click(170, 165)

// click multiple elements by passing multiple: true
cy.get('.action-labels>.label').click({ multiple: true })

// Ignore error checking prior to clicking
cy.get('.action-opacity>.btn').click({ force: true })

.click()

To check a checkbox or radio, you can use the .click() command.

// By default, .check() will check all
// matching checkbox or radio elements in succession, one after another
const checkboxes = await $$('.action-checkboxes [type="checkbox"]:not([disabled])')
for (const checkbox of checkboxes) {
  await checkbox.click()
  await expect(checkbox).toBeSelected()
}

const radios = await $$('.action-radios [type="radio"]:not([disabled])')
for (const radio of radios) {
  await radio.click()
  await expect(radio).toBeSelected()
}

// .click() can be used with a selector for a specific value
const radio1 = await $('.action-radios [value="radio1"]')
await radio1.click()
await expect(radio1).toBeSelected()

// .click() can be used in a loop for multiple values
const checkboxesToSelect = ['checkbox1', 'checkbox2']
for (const value of checkboxesToSelect) {
  const checkbox = await $(`.action-multiple-checkboxes [value="${value}"]`)
  await checkbox.click()
  await expect(checkbox).toBeSelected()
}

// Webdriver.io does not have a force option, you would need to enable the element first
const disabledCheckbox = await $('.action-checkboxes [disabled]')
await browser.execute((el) => el.disabled = false, disabledCheckbox)
await disabledCheckbox.click()
await expect(disabledCheckbox).toBeSelected()

const disabledRadio = await $('.action-radios [value="radio3"]')
await browser.execute((el) => el.disabled = false, disabledRadio)
await disabledRadio.click()
await expect(disabledRadio).toBeSelected()

.click()

To uncheck a checkbox or radio, you can use the .click() command.

// By default, .uncheck() will uncheck all matching
// checkbox elements in succession, one after another
const checkboxes = await $$('.action-check [type="checkbox"]:not([disabled])')
for (const checkbox of checkboxes) {
  if (await checkbox.isSelected()) {
    await checkbox.click()
  }
  await expect(checkbox).not.toBeSelected()
}

// Click to uncheck a specific checkbox
const checkbox1 = await $('.action-check [value="checkbox1"]')
if (!await checkbox1.isSelected()) {
  await checkbox1.click() // ensure it's checked first
}
await checkbox1.click() // uncheck it
await expect(checkbox1).not.toBeSelected()

// Click to uncheck multiple checkboxes
const checkboxesToUncheck = ['checkbox1', 'checkbox3']
for (const value of checkboxesToUncheck) {
  const checkbox = await $(`.action-check [value="${value}"]`)
  if (!await checkbox.isSelected()) {
    await checkbox.click() // ensure it's checked first
  }
  await checkbox.click() // uncheck it
  await expect(checkbox).not.toBeSelected()
}

// Webdriver.io does not have a force option, you would need to enable the element first
const disabledCheckboxToUncheck = await $('.action-check [disabled]')
await browser.execute((el) => el.disabled = false, disabledCheckboxToUncheck)
if (await disabledCheckboxToUncheck.isSelected()) {
  await disabledCheckboxToUncheck.click()
}
await expect(disabledCheckboxToUncheck).not.toBeSelected()

.selectByVisibleText()

To select an option in a select, use the .selectByVisibleText() command.

// at first, no option should be selected
const select = await $('.action-select')
await expect(select).toHaveValue('--Select a fruit--')

// Select option(s) with matching text content
const select = await $('.action-select')
await select.selectByVisibleText('apples')
await expect(select).toHaveValue('fr-apples')

const multiSelect = await $('.action-select-multiple')
await multiSelect.selectByVisibleText('apples')
await multiSelect.selectByVisibleText('oranges')
await multiSelect.selectByVisibleText('bananas')
const selectedValues = await multiSelect.getValue()
expect(selectedValues).toEqual(['fr-apples', 'fr-oranges', 'fr-bananas'])

// Select option(s) with matching value
await select.selectByAttribute('value', 'fr-bananas')
await expect(select).toHaveValue('fr-bananas')

await multiSelect.selectByAttribute('value', 'fr-apples')
await multiSelect.selectByAttribute('value', 'fr-oranges')
await multiSelect.selectByAttribute('value', 'fr-bananas')
const newSelectedValues = await multiSelect.getValue()
expect(newSelectedValues).toEqual(['fr-apples', 'fr-oranges', 'fr-bananas'])

// assert the selected values include oranges
const finalSelectedValues = await multiSelect.getValue()
expect(finalSelectedValues).toContain('fr-oranges')

.scrollIntoView()

To scroll an element into view, use the .scrollIntoView() command.

// normally all of these buttons are hidden,
// because they're not within
// the viewable area of their parent
// (we need to scroll to see them)
const button = await $('#scroll-horizontal button')
await expect(button).not.toBeDisplayed()

// scroll the button into view, as if the user had scrolled
const button = await $('#scroll-horizontal button')
await button.scrollIntoView()
await expect(button).toBeDisplayed()

const vButton = await $('#scroll-vertical button')
await expect(vButton).not.toBeDisplayed()

// Webdriver.io handles the scroll direction needed
await vButton.scrollIntoView()
await expect(vButton).toBeDisplayed()

const bothButton = await $('#scroll-both button')
await expect(bothButton).not.toBeDisplayed()

// Webdriver.io knows to scroll to the right and down
await bothButton.scrollIntoView()
await expect(bothButton).toBeDisplayed()

browser.scroll()

To scroll the window or a scrollable element to a specific position, use the browser.scroll() command.

// You can scroll to 9 specific positions of an element:
//  -----------------------------------
// | topLeft        top       topRight |
// |                                   |
// |                                   |
// |                                   |
// | left          center        right |
// |                                   |
// |                                   |
// |                                   |
// | bottomLeft   bottom   bottomRight |
//  -----------------------------------

// if you chain .scrollTo() off of cy, we will
// scroll the entire window
await browser.scroll(0, 10000) // scrolls to the bottom of the page

const hScroll = await $('#scrollable-horizontal')
await hScroll.scroll(10000, 0) // scroll to the right

// or you can scroll to a specific coordinate:
// (x axis, y axis) in pixels
const vScroll = await $('#scrollable-vertical')
await vScroll.scroll(250, 250)

// Webdriver.io does not support percentage scrolling directly, you would calculate it
const bothScroll = await $('#scrollable-both')
const size = await bothScroll.getSize()
await bothScroll.scroll(size.width * 0.75, size.height * 0.25)

// Easing is not supported in Webdriver.io's scroll command

// Duration is not supported in Webdriver.io's scroll command

browser.execute()

To trigger an event on a DOM element, you can use the browser.execute() command to dispatch an event.

// To interact with a range input (slider)
// we need to set its value & trigger the
// event to signal it changed

// Here, we invoke jQuery's val() method to set
// the value and trigger the 'change' event
const range = await $('.trigger-input-range')
await browser.execute((el) => {
  el.value = 25
  el.dispatchEvent(new Event('change'))
}, range)
const p = await range.next()
await expect(p).toHaveText('25')
